Poker Automation

I'm not going to click if I don't have to


For a while, my friends and I played a mobile poker game. Every four hours, you could log in and collect free chips to use, and I immediately saw the potential of having something collect my chips automatically for me. Here is my first attempt, a casper.js headless browser script that would log in and press the button for me. Unfortunately, I was met with a "cannot reach the server right now" message. Undeterred, I tried catching the request that was being sent to their servers. I found that by skipping the game entirely and just sending the request, they would add the chips to my game account. By setting up a simple cron job to curl their servers every four hours, I have been collecting my free chips for months.

var casper = require('casper').create({   
    verbose: true, 
    logLevel: 'debug',
    pageSettings: {
        loadImages:  true,         // The WebPage instance used by Casper will
        loadPlugins: true,         // use these settings
      userAgent: 'Mozilla/5.0 (compatible, MSIE 11, Windows NT 6.3; Trident/7.0; rv:11.0) like Gecko',
    }
});

casper.start('http://www.facebook.com', function() {
   this.fill('form#login_form', { 
        email: email, 
        pass:  password
    }, true);
});

casper.thenOpen('http://playwsop.com/?utm_source=wsopcom&utm_medium=weblink_WSOPPoker_MobnSoc&utm_campaign=wsopcom_mobnSoc_weblink', function() {
  this.capture('image2.png', {
        top: 0,
        left: 0,
        width: 1920,
        height: 1080
    });
});

var linkSelector = 'div.lpcPText';

casper.then(function() {
    if (!this.exists(linkSelector)) return;
    this.click(linkSelector);
    this.evaluate(function(linkSelector) {
        __utils__.findOne(linkSelector).setAttribute("className", "clicked");
    }, linkSelector);
});

casper.wait('15000', function() {
  this.capture('image3.png', {
        top: 0,
        left: 0,
        width: 1920,
        height: 1080
    });

});

casper.run();
